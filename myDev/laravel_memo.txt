//テーブル作成
php artisan make:migration テーブル名
+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
//migration 実行
php artisan migrate
+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+



//ロールバック
+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
# 最後に「一度に」実行したマイグレーションをまとめて元に戻す.
$ php artisan migrate:rollback

# 巻き戻す数を指定する.
php artisan migrate:rollback --step=5

# 全部のマイグレーションを戻す.
php artisan migrate:reset
